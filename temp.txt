enum State{
    ON, OFF
};

class Batti{
    private:
    int onTime;
    int offTime;
    int pin;
    long lastTime;
    State state;
    public:
    void blink(int onTime, int offTime, int pin){
        this->onTime = onTime;
        this->offTime = offTime;
        this->pin = pin;
        pinMode(pin, OUTPUT);
        digitalWrite(pin, HIGH);
        state = ON;
    }
    void ledoop(){
        if(state==ON){
            if((millis() - lastTime) >= onTime){
                digitalWrite(pin, LOW);
                state = OFF;
                lastTime = millis();
            }
        }else if(state==OFF){
            if((millis() - lastTime) >= offTime){
                digitalWrite(pin, HIGH);
                state = ON;
                lastTime = millis();
            }
        }
    }

    void reset(){
        state = (State)-1;
        digitalWrite(LED_BUILTIN, HIGH);
    }
};
